```html

<!DOCTYPE html>

<html>

<head>

  <title>Raspberry Pi Pico Geolocation</title>

</head>

<body>

  <h1>Raspberry Pi Pico Geolocation</h1>

  <div id="location">Loading...</div>


  <script>

    // Получение геолокации с сервера

    fetch('/location')

      .then(response => response.json())

      .then(data => {

        const locationDiv = document.getElementById('location');

        locationDiv.textContent = `Latitude: ${data.latitude}, Longitude: ${data.longitude}`;

      })

      .catch(error => {

        console.error('Error fetching location:', error);

        locationDiv.textContent = 'Error loading location';

      });

  </script>

</body>

</html>

``
